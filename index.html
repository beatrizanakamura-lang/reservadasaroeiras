const video = document.getElementById('video'); const btn = document.getElementById('btnFoto'); const countdown = document.getElementById('countdown'); const preview = document.getElementById('preview'); const previewImg = document.getElementById('previewImg'); const flash = document.getElementById('flash'); let fotoFinal = ""; /* INICIAR CÂMERA */ navigator.mediaDevices.getUserMedia({ video: { facingMode: "environment" } }) .then(stream => { video.srcObject = stream; }) .catch(err => { alert("Erro ao acessar câmera: " + err.message); }); /* BOTÃO (100% FUNCIONAL) */ btn.addEventListener("click", () => { iniciarContagem(); }); /* CONTAGEM */ function iniciarContagem() { let tempo = 3; countdown.style.display = "block"; countdown.innerText = tempo; const intervalo = setInterval(() => { tempo--; countdown.innerText = tempo; if (tempo <= 0) { clearInterval(intervalo); countdown.style.display = "none"; tirarFoto(); } }, 1000); } /* TIRAR FOTO */ function tirarFoto() { try { const canvas = document.createElement('canvas'); canvas.width = video.videoWidth; canvas.height = video.videoHeight; const ctx = canvas.getContext('2d'); /* FLASH */ flash.style.opacity = 1; setTimeout(() => flash.style.opacity = 0, 150); /* FOTO */ ctx.drawImage(video, 0, 0); const moldura = new Image(); moldura.src = 'MOLDURA.png'; moldura.onload = () => { ctx.drawImage(moldura, 0, 0, canvas.width, canvas.height); fotoFinal = canvas.toDataURL("image/png"); previewImg.src = fotoFinal; preview.style.display = "block"; }; moldura.onerror = () => { alert("Erro ao carregar moldura (verifique o nome MOLDURA.png)"); }; } catch (e) { alert("Erro ao capturar imagem: " + e.message); } } /* SALVAR + COMPARTILHAR */ function salvar() { const link = document.createElement('a'); link.download = "reserva-aroeiras.png"; link.href = fotoFinal; link.click(); compartilhar(); } /* COMPARTILHAMENTO NATIVO */ function compartilhar() { fetch(fotoFinal) .then(res => res.blob()) .then(blob => { const file = new File([blob], "reserva-aroeiras.png", { type: "image/png" }); if (navigator.canShare && navigator.canShare({ files: [file] })) { navigator.share({ files: [file], title: "Reserva das Aroeiras", text: "Vivendo a experiência Reserva das Aroeiras ✨" }); } }) .catch(() => {}); } /* ABRIR INSTAGRAM + COPIAR TEXTO */ function abrirInstagram() { const texto = "Vivendo a experiência Reserva das Aroeiras ✨\n\n@reservadasaroeirascuiaba\n#ReservaDasAroeiras #AltoPadrao #Cuiaba"; navigator.clipboard.writeText(texto); /* tenta abrir app */ window.location.href = "instagram://camera"; /* fallback */ setTimeout(() => { window.open("https://www.instagram.com/", "_blank"); }, 1000); alert("Legenda copiada! Agora é só colar no Instagram 📸"); } /* REFAZER */ function refazer() { preview.style.display = "none"; }
